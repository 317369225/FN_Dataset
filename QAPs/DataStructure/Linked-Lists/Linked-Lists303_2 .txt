This could be easily found on the internet as well. public class Solution { public void printReverseList(ListNode head) { if (head != null) { printReverseList(head.next); System.out.print(head.data + " "); } } public static void main(String args[]) { /* Creating a linked list -- 1 -> 2 -> 3 -> 4 -> 5 */ ListNode head = new ListNode(1); head.next = new ListNode(2); head.next.next = new ListNode(3); head.next.next.next = new ListNode(4); head.next.next.next.next = new ListNode(5); Solution sol = new Solution(); /* Call the print reverse method */ sol.printReverseList(head); } } class ListNode { int data; ListNode next; public ListNode(int data) { this.data = data; this.next = null; } } The linked list we built is 1 -> 2 -> 3 -> 4 -> 5 and the output is 5 -> 4 -> 3 -> 2 -> 1.
