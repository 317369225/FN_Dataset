Standard linear solution exists for solving this problem: private int Partition(int[] a, int p, int r) { int x = a[r]; int i = p - 1; int temp = 0; for (int j = p; j < r; j++) { if (a[j] <= x) { i++; temp = a[i]; a[i] = a[j]; a[j] = temp; } } temp = a[i + 1]; a[i + 1] = a[r]; a[r] = temp; return (i + 1); } source: QuickSort partition algorithm 149 Views  Answer requested by 1 person
