Can a class have array of objects?Yes, you can. You must be learning to program. Below is a sample C++ code: #include <iostream> #include <vector> using namespace std;  // Cls1 & Cls2 are classes. // Cls1 objects are stored as array (vector) inside Cls2. // main function prints the array of Cls1 objects, stored in Cls2.  class Cls1 { public: int n; Cls1(int n1) { n = n1; } };  class Cls2 { std::vector<Cls1*> array_of_cls1_objects; public: Cls2() { Cls1 *p1 = new Cls1(11); Cls1 *p2 = new Cls1(22);  array_of_cls1_objects.push_back(p1); array_of_cls1_objects.push_back(p2); } void print_cls1_array_of_objects() { for (unsigned int i = 0; i < array_of_cls1_objects.size(); i++) { cout << array_of_cls1_objects[i]->n << endl; // retrieve the array through pointer } } };  void main() { Cls2 o; o.print_cls1_array_of_objects(); }   // Output of above program is below: // // 11 // 22 // All the best in learning. 321 Views  View Upvotes
