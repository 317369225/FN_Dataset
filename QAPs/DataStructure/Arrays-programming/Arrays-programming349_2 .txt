How do I display top 5 numbers in an array in c++?Here a simple way, using vector and sorting methods: #include<iostream> #include<algorithm> #include<vector>  using namespace std;  int main() { int list[] = {90,44,55,18,92,12,16,20,88,48} vector<int> myvector (list, list+10); int tam = myvector.size();  sort(myvector.begin(),myvector.end()); for (int i = 0; i < 5; i++) { cout << "number: " << myvector[tam - 1 - i] << endl; }  } The output here: number: 92 number: 90 number: 88 number: 55 number: 48 272 Views
