How do I make all -ve numbers one side and +ve another?import java.util.*; public class Solution{  public static void main(String []args){ Scanner sc = new Scanner(system.in); int t = sc.nextInt(); for(int i=0; i <t; ++i){ int n = sc.nextInt(); int[] arr = new int[n]; for(int j =0; j < n; ++j) arr[j] = sc.nextInt(); int swap = 0; for(int s =0, e = n -1; s < e; ){ while(arr[s] < 0 && s < e) ++s; while(arr[e] >0 && e > s) --e; if (arr[s] > 0 && arr[e] < 0){ swap++; int temp = arr[s]; arr[s] = arr[e]; arr[e] = temp; ++s; --e; } } System.out.println(swap); } sc.close(); } } 72 Views
