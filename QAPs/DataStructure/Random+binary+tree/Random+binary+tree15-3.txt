How are input files generated for problems on different online judges by problem setter?
There can be multiple test files for a problem and the generator for one of the test files looks similar to this. #include<iostream>#include<cstdlib>using namespace std;int R, C;char getRand(){    int val = rand()%2;    if(val)    return 'a'+rand()%2;    return 'u'+rand()%6;}int main(){    int t,i,j;    freopen("input.txt","w",stdout);    t=5;    cout<<t<<"n";    while(t--){        R = 2+rand()%9;        C = 2+rand()%9;        cout<<R<<" "<<C<<"n";        for(i=0;i<R;i++){            for(j=0;j<C;j++){                cout<<getRand();            }            cout<<'n';        }    }    return 0;} This generates a small input for the SPOJ problem KATHTHI. Note that you can tweak the random function to however you want.
