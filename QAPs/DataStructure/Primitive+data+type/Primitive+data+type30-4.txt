In java you can now do this kind of stuff : package be.programaths;import java.time.Duration;import java.time.Instant;import java.util.stream.IntStream;public class Foo {    public static void main(String[] args) {        impl1();        impl2();    }    public static void impl1(){        Instant start = Instant.now();        StringBuilder sb=new StringBuilder();        for(int i=0;i<100000;i++){            if(i%97>i%131 && i%927==0){                sb.append(",").append(i);            }        }        sb.deleteCharAt(0);        System.out.println(sb.toString());        Instant end = Instant.now();        System.out.println(Duration.between(start, end));    }    public static void impl2(){        Instant start = Instant.now();        String StringlistOfInt=IntStream                .rangeClosed(1, 100000)                .filter(n -> n%97>n%131 && n%927==0)                .mapToObj(String::valueOf)                .reduce("",(a, b) -> a + "," + b)                .substring(1);        System.out.println(StringlistOfInt);        Instant end = Instant.now();        System.out.println(Duration.between(start, end));    }} What it does is creating a huge list of number and filter them to output only the one which match a given condition. impl1 and impl2 have the same result, but they are written in a totally different style! Java is all about higher level language. You don't have to manage memory as much than in C and C++. Also, most things are "safe". No wild pointers.
