This would work: public static void main(String [] args) { char [] k = {'a', 'x', 'z', 'y', 'q'}; boolean isChanged = true;  // sorts a char array k using bubble sort while(isChanged){ isChanged = false; for (int i = 0; i < k.length; i++) { char temp;  for (int j = i; j < k.length -1; j++) {  if (k[j] > k[j + 1]) {  temp = k[j];  k[j] = k[j + 1];  k[j + 1] = temp;  }//end if block  }//end int j for loop  }//end int n for loop }//end while loop }//end main you also have to include the while loop because you want to keep bubble sorting until all elements have been sorted.
