What is the next number in this sequence: 1, 11, 21, 1211, 111221, 312211, 13112221?
This C++ program prints the sequence, which grows exponentially in length: #include <iostream>#include <sstream>using namespace std;int main(){string x="1";while(true){   int size=x.length();  string s="";  int i=0;           while(i<size)  {    int count=1;    while((i+1)<size&&x[i]==x[i+1])    {     count++;     i++;         }    ostringstream convert;    convert<<count;        s=s+convert.str()+x[i];    i++;  }            cout<<s<<endl; x=s;system("pause");}}    For the record: This question was asked in Facebook recruitment interview.
