How do I find the first non repeated character of a given string?
Since the index positions can never be non-negative therefore we can use -9999 to denote characters that are non-present in the string (the array will be initialized to -9999) and -1 to denote characters already encountered. The positive values will denote the indexes of the non-repeating characters. Loop through the array for the minimum positive value would give us the index of the first non-repeating character in the string. [code] /** * Created by fsyeda on 5/7/15. */ public class FirstNonRepeatingChar { public static char findFirstNonRepeatingChar(String str) { final int NON_EXISTENT = -9999; int [] alph = new int[26]; // initialize array to NON_EXISTENT         for (int i = 0; i < alph.length; i++) alph[i] = NON_EXISTENT; // if it's occurring first time: store the index         // if it has a NON(NON-EXISTENT) value then store the index as -1 --> meaning that it has already occurred         for (int i = 0; i < str.length(); i++) { if (alph[str.charAt(i) - 'a'] == NON_EXISTENT) alph[str.charAt(i) - 'a'] = i; else if (alph[str.charAt(i) - 'a'] != -1) alph[str.charAt(i) - 'a'] = -1;         } // at this point the array has:         // indexes of non-repeating characters         // non-repeating characters indexed as NON_EXISTENT         // repeating characters indexed as -1         // loop through the array to filter out NON_EXISTENT and REPEATING char (condition: alph[i] >= 0) and keep tracking of min index seen         int min = Integer.MAX_VALUE; for (int i = 0; i < alph.length; i++) { if (alph[i] >= 0 && alph[i] < min) min = alph[i];         } return (min < str.length())? str.charAt(min) : '\n';     } public static void main(String [] args) {         System.out.println('l' == findFirstNonRepeatingChar("google"));         System.out.println('l' == findFirstNonRepeatingChar("linkedin"));         System.out.println('m' == findFirstNonRepeatingChar("amazon"));         System.out.println('r' == findFirstNonRepeatingChar("ritz"));         System.out.println('t' == findFirstNonRepeatingChar("brbzrzt"));         System.out.println('f' == findFirstNonRepeatingChar("geeksforgeeks"));         System.out.println('\n' == findFirstNonRepeatingChar("abarbcrc"));     } } [/code]
