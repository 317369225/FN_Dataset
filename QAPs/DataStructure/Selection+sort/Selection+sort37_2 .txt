First find the minimum of array, put/swap it with the value at its position. Now increase the array indices and go to step 1.  1 2 3 4 5 6 7 8 91011121314151617181920212223242526 int min_idx(vector<int> arr, int start, int end) {  int min = start;  for(int i = start+1; i < end; i++)  {  if(arr[min] > arr[i])  min = i;  }  return min; } void swap_sort(vector<int>& data) {  int len = data.size();  int count = 0; // to keep the number of swap performed.  for(int i = 0; i < len; i++)  {  int temp = min_idx(data,i,len);  if(temp != i)  {  swap(data[i],data[temp]);  count++;  }  }  cout<<"Printing total number of swap performed = "<<count<<"\n";  // You can check that the resultant array will be sorted in ascending order. }
