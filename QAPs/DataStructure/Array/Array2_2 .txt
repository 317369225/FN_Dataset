Here's an example of an infinite prime generator I just whipped up. Warning: I haven't done this in Java before, just adapted one I have in Python, so I can't guarantee that it's perfect, but the general gist should be right.  1 2 3 4 5 6 7 8 91011121314151617181920212223242526272829303132333435 import java.math.BigInteger; import java.util.Iterator; public class Test { 	public static class PrimeGenerator implements Iterable<BigInteger> { 		@Override 		public Iterator<BigInteger> iterator() { 			return new Iterator<BigInteger>() { 				private BigInteger i, one = i = BigInteger.valueOf(1); 				@Override 				public boolean hasNext() { 					return true; 				} 				@Override 				public BigInteger next() { 					for (i = i.add(one); !i.isProbablePrime(10); i = i.add(one)); 					return i; 				} 				@Override 				public void remove() { 				} 			}; 		} 	} 	public static void main(String[] args) { 		for (BigInteger i : new PrimeGenerator()) { 			System.out.println(i); 		} 	} }
